```python
T = int(input())
for t in range(T):
    N, P = map(int, input().split())
    ilist = list(map(int,input().split()))
    cnt = 0
    
    # 최댓값을 찾기 위해 정렬
    sorted_list = sorted(ilist)

   
    while True:
        # 인덱스 값을 통해 순서를 알기 원하는 값을 지정
        if P < 0:
            P += len(ilist)
            
        # 제일 앞의 값이 최대값이라면 cnt를 1 증가하고
        # sorted_list에서 마지막 값을 ilist에서 제일 앞의 값을 제거
        # P -= 1을 통해 P를 옮겨줌
        # 제일 앞의 값이 최대값일 때 P의 값이 0이라면 반복을 탈출하고 cnt를 출력
        # 그렇지 않을 시 계속 반복
        if ilist[0] == sorted_list[-1]:
            
            cnt += 1
            if P == 0:
                break
            sorted_list.pop()
            ilist.pop(0)
            P -= 1
        else:
            ilist.append(ilist.pop(0))
            P -= 1

    print(cnt)
```



## 5397

```python
T = int(input())
for t in range(T):
    input_lst = list(input())
    temp_lst = []
    result = []
    for i in input_lst:
        if i == '<':
            if result:
                temp_lst.append(result.pop())
            else:
                continue
        elif i == '>':
            if temp_lst:
                result.append(temp_lst.pop())
            else:
                continue
        elif i == '-':
            if result:
                result.pop()
            else:
                continue
            
        else:
            result.append(i)
    temp_lst.reverse()
    for x in temp_lst:
        result.append(x)
    print(''.join(result))
```

## 1074 -> 메모리 초과가 나오긴 했지만 잘 풀었따 -> 내일 다시 한번 풀어보기

```python
N, r, c = map(int, input().split())

cnt = 0

def Z_funtion(arr, cnt, x):
    if len(arr) == 2:
        for i in range(2):
            for j in range(2):
                if arr[i][j] == 'x':
                    x = cnt
                arr[i][j] = cnt
                cnt += 1

        return arr, cnt, x

    else:
        # 4등분한 배열
        res = []
        for i in range(2):
            for j in range(2):
                tmp = []
                for row in arr[len(arr)//2*i:len(arr)//2*(i+1)]:
                    tmp.append(row[len(arr)//2*j:len(arr)//2*(j+1)])
                z_val = Z_funtion(tmp, cnt, x)
                res.append(z_val[0])
                cnt = z_val[1]
                x = z_val[2]
                if x != 0:
                    return res, cnt, x
        return res, cnt, x


cnt = 0
x = 0
N_arr = [[0]*(2**N) for _ in range(2**N)]
N_arr[r][c] = 'x'
print(Z_funtion(N_arr, cnt, x)[2])
```
